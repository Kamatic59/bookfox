// Twilio SMS Webhook Handler v2 - No JWT Required
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts';
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
  'Access-Control-Allow-Methods': 'POST, OPTIONS',
};

serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL') ?? '',
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? ''
    );

    const TWILIO_ACCOUNT_SID = Deno.env.get('TWILIO_ACCOUNT_SID') ?? '';
    const TWILIO_AUTH_TOKEN = Deno.env.get('TWILIO_AUTH_TOKEN') ?? '';
    const GEMINI_API_KEY = Deno.env.get('GEMINI_API_KEY') ?? '';

    // Parse form data from Twilio
    const formData = await req.formData();
    const params: Record<string, string> = {};
    formData.forEach((value, key) => {
      params[key] = value.toString();
    });

    console.log('SMS webhook received:', JSON.stringify(params));

    const toPhone = params.To || '';
    const fromPhone = params.From || '';
    const messageBody = params.Body || '';
    const messageSid = params.MessageSid || '';

    // Find business
    const { data: business, error: bizError } = await supabase
      .from('businesses')
      .select('*')
      .eq('twilio_phone', toPhone)
      .single();

    if (bizError || !business) {
      console.error('Business not found:', toPhone);
      return new Response('<?xml version="1.0" encoding="UTF-8"?><Response></Response>', {
        headers: { ...corsHeaders, 'Content-Type': 'application/xml' },
      });
    }

    // Get AI settings
    const { data: aiSettings } = await supabase
      .from('ai_settings')
      .select('*')
      .eq('business_id', business.id)
      .single();

    // Find or create conversation
    let { data: conversation } = await supabase
      .from('conversations')
      .select('*')
      .eq('business_id', business.id)
      .eq('customer_phone', fromPhone)
      .eq('status', 'active')
      .maybeSingle();

    if (!conversation) {
      // Create lead
      const { data: lead } = await supabase
        .from('leads')
        .insert({ business_id: business.id, phone: fromPhone, source: 'sms', status: 'new' })
        .select()
        .single();

      if (lead) {
        const { data: newConvo } = await supabase
          .from('conversations')
          .insert({
            business_id: business.id,
            lead_id: lead.id,
            business_phone: toPhone,
            customer_phone: fromPhone,
            status: 'active',
            mode: 'ai',
            ai_context: { collected_info: {} },
          })
          .select()
          .single();
        conversation = newConvo;
      }
    }

    if (!conversation) {
      console.error('Failed to get/create conversation');
      return new Response('<?xml version="1.0" encoding="UTF-8"?><Response></Response>', {
        headers: { ...corsHeaders, 'Content-Type': 'application/xml' },
      });
    }

    // Store inbound message
    await supabase.from('messages').insert({
      conversation_id: conversation.id,
      direction: 'inbound',
      content: messageBody,
      sender_type: 'customer',
      twilio_sid: messageSid,
    });

    // Skip AI if in human mode
    if (conversation.mode === 'human') {
      console.log('Human mode - skipping AI');
      return new Response('<?xml version="1.0" encoding="UTF-8"?><Response></Response>', {
        headers: { ...corsHeaders, 'Content-Type': 'application/xml' },
      });
    }

    // Get message history
    const { data: messages } = await supabase
      .from('messages')
      .select('content, sender_type')
      .eq('conversation_id', conversation.id)
      .order('created_at', { ascending: true })
      .limit(10);

    // Build AI prompt
    const assistantName = aiSettings?.assistant_name || 'BookFox';
    const systemPrompt = `You are ${assistantName}, a friendly AI assistant for ${business.name}.
Help customers via SMS. Be warm but brief (1-2 sentences). 
Gather: what service they need, urgency, home or business.
Respond with ONLY the message text - no JSON, no formatting.`;

    const chatHistory = (messages || []).map(m => ({
      role: m.sender_type === 'customer' ? 'user' : 'model',
      parts: [{ text: m.content }],
    }));

    // Call Gemini
    const geminiResponse = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_API_KEY}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          systemInstruction: { parts: [{ text: systemPrompt }] },
          contents: chatHistory,
          generationConfig: { temperature: 0.7, maxOutputTokens: 150 },
        }),
      }
    );

    let aiText = "Thanks for your message! Someone from our team will get back to you soon.";
    
    if (geminiResponse.ok) {
      const geminiData = await geminiResponse.json();
      aiText = geminiData.candidates?.[0]?.content?.parts?.[0]?.text || aiText;
      aiText = aiText.trim().slice(0, 320);
    } else {
      console.error('Gemini error:', await geminiResponse.text());
    }

    // Send SMS response
    const twilioUrl = `https://api.twilio.com/2010-04-01/Accounts/${TWILIO_ACCOUNT_SID}/Messages.json`;
    const auth = btoa(`${TWILIO_ACCOUNT_SID}:${TWILIO_AUTH_TOKEN}`);

    const smsResponse = await fetch(twilioUrl, {
      method: 'POST',
      headers: {
        'Authorization': `Basic ${auth}`,
        'Content-Type': 'application/x-www-form-urlencoded',
      },
      body: new URLSearchParams({
        To: fromPhone,
        From: toPhone,
        Body: aiText,
      }),
    });

    const smsData = await smsResponse.json();
    console.log('SMS response sent:', smsData.sid);

    // Store outbound message
    await supabase.from('messages').insert({
      conversation_id: conversation.id,
      direction: 'outbound',
      content: aiText,
      sender_type: 'ai',
      twilio_sid: smsData.sid,
      twilio_status: smsData.status,
    });

    return new Response('<?xml version="1.0" encoding="UTF-8"?><Response></Response>', {
      headers: { ...corsHeaders, 'Content-Type': 'application/xml' },
    });

  } catch (error) {
    console.error('SMS webhook error:', error);
    return new Response('<?xml version="1.0" encoding="UTF-8"?><Response></Response>', {
      headers: { ...corsHeaders, 'Content-Type': 'application/xml' },
    });
  }
});
